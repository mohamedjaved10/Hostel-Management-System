Resolution within 3 days." }; complaints.push( complaint ); saveComplaints(); addActivityLog( "Complaint Submitted : " + complaint.id ); const responseContainer = document.getElementById( "complaintResponseContainer" ); if (responseContainer) { responseContainer.innerHTML = `

Complaint Submitted Successfully

Complaint ID: ${complaint.id}

Status: ${complaint.status}

Save your Complaint ID for future tracking.

`; } alert( "Complaint Submitted Successfully" ); complaintForm.reset(); } ); } // ===================================== // LOAD COMPLAINT HISTORY // ===================================== function loadComplaintHistory() { const historyContainer = document.getElementById( "complaintHistoryContainer" ); if (!historyContainer) { return; } if (complaints.length === 0) { historyContainer.innerHTML = `

No Complaints Submitted Yet

`; return; } let html = ""; complaints.forEach( function(complaint) { html += `

${complaint.id}

Category: ${complaint.category}

Status: ${complaint.status}

`; } ); historyContainer.innerHTML = html; } loadComplaintHistory(); // ===================================== // VIEW COMPLAINT DETAILS // ===================================== function viewComplaint( complaintId ) { const complaint = complaints.find( item => item.id === complaintId ); if (!complaint) { return; } const detailsContainer = document.getElementById( "complaintDetailsContainer" ); if (detailsContainer) { detailsContainer.innerHTML = `

Complaint Details

Complaint ID: ${complaint.id}

Student Name: ${complaint.studentName}

Room Number: ${complaint.roomNumber}

Category: ${complaint.category}

Priority: ${complaint.priority}

Status: ${complaint.status}

`; } const remarksContainer = document.getElementById( "adminRemarksContainer" ); if (remarksContainer) { remarksContainer.innerHTML = `

${complaint.remarks}

`; } } // ===================================== // TRACK COMPLAINT // ===================================== const trackComplaintForm = document.getElementById( "trackComplaintForm" ); if (trackComplaintForm) { trackComplaintForm.addEventListener( "submit", function(event) { event.preventDefault(); const complaintId = document.getElementById( "trackComplaintId" ).value.trim(); const complaint = complaints.find( item => item.id === complaintId ); const resultContainer = document.getElementById( "trackComplaintResult" ); if (!resultContainer) { return; } if (!complaint) { resultContainer.innerHTML = `

Complaint Not Found

`; return; } resultContainer.innerHTML = `

Complaint Found

Complaint ID: ${complaint.id}

Status: ${complaint.status}

`; viewComplaint( complaint.id ); } ); }
